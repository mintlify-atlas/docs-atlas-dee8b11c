Layout components provide the structure and navigation for the Codante.io application.

## RootLayout (AppLayout)

Main application layout that wraps page content with navbar and footer.

### Props

<ParamField path="children" type="React.ReactNode" required>
  Page content to render
</ParamField>

<ParamField path="user" type="User | null" required>
  Current authenticated user or null if not logged in
</ParamField>

<ParamField path="hideNavbarLinks" type="boolean" default="false">
  Hide navigation links (for minimal layouts)
</ParamField>

### Usage

```tsx
import AppLayout from "~/components/_layouts/root-layout";
import type { User } from "~/lib/models/user.server";

export default function MyPage() {
  const { user } = useOutletContext<{ user: User | null }>();
  
  return (
    <AppLayout user={user}>
      <div className="container">
        {/* Your page content */}
      </div>
    </AppLayout>
  );
}
```

### Structure

The layout renders:
1. Upper alert banner portal (`#alert-banner-upper`)
2. Navbar component
3. Lower alert banner portal (`#alert-banner-lower`)
4. Main content area with minimum height
5. Footer component

```tsx
<>
  <div id="alert-banner-upper"></div>
  <Navbar user={user} hideLinks={hideNavbarLinks} />
  <div id="alert-banner-lower"></div>
  <main className="lg:py-10 py-6 min-h-[calc(100vh-170px)] mx-auto">
    {children}
  </main>
  <Footer />
</>
```

## Navbar

### Props

<ParamField path="user" type="User | null" required>
  Current authenticated user or null
</ParamField>

<ParamField path="hideLinks" type="boolean" default="false">
  Hide navigation links
</ParamField>

### Animations

The navbar uses Framer Motion for smooth animations:
- Animated hamburger icon that transforms to X
- Slide-in mobile menu with fade effect
- Sound effects on menu interactions

### Dark Mode

The navbar automatically adapts to the current color mode:
```tsx

{colorMode === "light" ? (
  <img src="/cdnt-light.svg" alt="Codante" />

  <img src="/cdnt.svg" alt="Codante" />

```

## Layout Patterns

### Standard Page Layout

```tsx
import { useOutletContext } from "react-router";
import AppLayout from "~/components/_layouts/root-layout";
import type { User } from "~/lib/models/user.server";

export default function StandardPage() {
  const { user } = useOutletContext<{ user: User | null }>();
  
  return (
    <AppLayout user={user}>
      <div className="container mx-auto px-4">
        <h1>Page Title</h1>
        {/* Page content */}
      </div>
    </AppLayout>
  );
}
```

### Minimal Layout (No Nav Links)

```tsx
export default function PlayerPage() {
  const { user } = useOutletContext<{ user: User | null }>();
  
  return (
    <AppLayout user={user} hideNavbarLinks={true}>
      <div className="w-full">
        {/* Full-width video player */}
      </div>
    </AppLayout>
  );
}
```

### Layout with Alert Banners

```tsx
import { useEffect } from "react";
import { createPortal } from "react-dom";

export default function PageWithAlert() {
  const { user } = useOutletContext<{ user: User | null }>();
  
  return (
    <>
      {createPortal(
        <div className="bg-brand-500 text-white p-4 text-center">
          Important announcement!
        </div>,
        document.getElementById('alert-banner-upper')!
      )}
      
      <AppLayout user={user}>
        {/* Page content */}
      </AppLayout>
    </>
  );
}
```

## Authentication Integration

Layout components integrate with the authentication system:

```tsx
// In loader
import { requireAuth } from "~/lib/services/auth.server";

export async function loader({ request }: LoaderFunctionArgs) {
  const user = await requireAuth(request);
  return { user };
}

// In component
const { user } = useOutletContext<{ user: User }>();

<AppLayout user={user}>
  {/* Authenticated content */}
</AppLayout>
```

For guest-only pages:
```tsx
import { requireGuest } from "~/lib/services/auth.server";

export async function loader({ request }: LoaderFunctionArgs) {
  await requireGuest(request);
  return { user: null };
}
```
