## Overview

The Challenge model handles all functionality related to mini-projects ("mini-projetos") on Codante.io, including challenge data, user participation, submissions, and progress tracking.

## Type Definitions

### ChallengeDifficulty

```typescript

```

### ChallengeCategory

## Usage Examples

### Fetching and Displaying Challenges

```typescript
import { getChallenges } from "~/lib/models/challenge.server";

export async function loader({ request }: LoaderFunctionArgs) {
  const url = new URL(request.url);
  const tech = url.searchParams.get("tecnologia") || "";
  const difficulty = url.searchParams.get("dificuldade") || "";

  const { challenges, featuredChallenge, totalChallenges } = await getChallenges({
    filters: { tecnologia: tech, dificuldade: difficulty },
    request,
  });

  return { challenges, featuredChallenge, totalChallenges };
}
```

### Enrolling in a Challenge

```typescript
import { joinChallenge } from "~/lib/models/challenge.server";

export async function action({ request, params }: ActionFunctionArgs) {
  const { slug } = params;
  const result = await joinChallenge({ slug, request });
  
  if (result.error) {
    return json({ error: result.error }, { status: 400 });
  }
  
  return json({ success: result.success });
}
```

### Submitting a Challenge

```typescript
import { submitChallenge } from "~/lib/models/challenge.server";

export async function action({ request, params }: ActionFunctionArgs) {
  const formData = await request.formData();
  const submissionUrl = formData.get("url") as string;
  const { slug } = params;

  const result = await submitChallenge(
    request,
    slug,
    submissionUrl,
  );

  return json(result);
}
```
