## Overview

The Track model handles learning paths ("trilhas") on Codante.io, which are structured collections of workshops, challenges, and other educational content organized into sections.

## Functions

### getTracks

Fetches all available tracks.

```typescript
async function getTracks(): Promise<Array<Track>>;
```

**Returns**: Array of all tracks

**Example**:
```typescript
import { getTracks } from "~/lib/models/track.server";

export async function loader() {
  const tracks = await getTracks();
  return json({ tracks });
}
```

### getTrack

Fetches a single track by slug.

<ParamField path="slug" type="string" required>
  Track slug identifier
</ParamField>

<ParamField path="request" type="Request" required>
  Request object for authentication
</ParamField>

```typescript
async function getTrack(
  slug: string,
  request: Request,
): Promise<Track>;
```

**Returns**: Track object or null if not found

**Example**:
```typescript
import { getTrack } from "~/lib/models/track.server";

export async function loader({ params, request }: LoaderFunctionArgs) {
  const track = await getTrack(params.slug, request);
  
  if (!track) {
    throw new Response("Track not found", { status: 404 });
  }
  
  return json({ track });
}
```

### toggleTrackableCompleted

Toggles completion status for a trackable item.

<ParamField path="trackableId" type="string" required>
  Trackable item identifier
</ParamField>

<ParamField path="request" type="Request" required>
  Request object for authentication
</ParamField>

```typescript
async function toggleTrackableCompleted(
  trackableId: string,
  request: Request,
): Promise<{
  success?: string;
  error?: string;
  timestamp?: string;
  trackableId?: string;
  completed?: boolean;
}>;
```

**Returns**: Success or error object with completion status

**Example**:
```typescript
import { toggleTrackableCompleted } from "~/lib/models/track.server";

export async function action({ request }: ActionFunctionArgs) {
  const formData = await request.formData();
  const trackableId = formData.get("trackableId") as string;
  
  const result = await toggleTrackableCompleted(trackableId, request);
  
  if (result.error) {
    return json({ error: result.error }, { status: 400 });
  }
  
  return json(result);
}
```

### isWorkshopTrackable

Type guard function to determine if a trackable is a workshop.

<ParamField path="workshop" type="WorkshopTrackable | ChallengeTrackable" required>
  Trackable item to check
</ParamField>

```typescript
function isWorkshopTrackable(
  workshop: WorkshopTrackable | ChallengeTrackable,
): workshop is WorkshopTrackable;
```

**Returns**: `true` if the trackable is a workshop, `false` otherwise

**Example**:
```typescript
import { isWorkshopTrackable } from "~/lib/models/track.server";

function renderTrackable(trackable: WorkshopTrackable | ChallengeTrackable) {
  if (isWorkshopTrackable(trackable)) {
    return <WorkshopItem workshop={trackable} />;
  } else {
    return <ChallengeItem challenge={trackable} />;
  }
}
```

## Usage Examples

### Track Listing Page

```typescript
import { getTracks } from "~/lib/models/track.server";

export async function loader() {
  const tracks = await getTracks();
  return json({ tracks });
}

export default function TracksPage() {
  const { tracks } = useLoaderData<typeof loader>();
  
  return (
    <div>
      <h1>Learning Paths</h1>
      {tracks.map((track) => (
        <div key={track.id}>
          <h2>{track.name}</h2>
          <p>{track.short_description}</p>
          <p>Difficulty: {track.difficulty}</p>
          <p>Duration: {track.duration_in_minutes} minutes</p>
        </div>
      ))}
    </div>
  );
}
```

### Track Detail with Progress

```typescript
import { getTrack } from "~/lib/models/track.server";
import { requireAuth } from "~/lib/services/auth.server";

export async function loader({ params, request }: LoaderFunctionArgs) {
  await requireAuth(request);
  const track = await getTrack(params.slug, request);
  
  if (!track) {
    throw new Response("Not Found", { status: 404 });
  }
  
  return json({ track });
}

export default function TrackDetailPage() {
  const { track } = useLoaderData<typeof loader>();
  
  return (
    <div>
      <h1>{track.name}</h1>
      <p>{track.description}</p>
      <div>
        {track.trackables.map((trackable) => (
          <div key={trackable.id}>
            <h3>{trackable.name}</h3>
            {trackable.type === "workshop" && (
              <p>Instructor: {trackable.instructor.name}</p>
            )}
          </div>
        ))}
      </div>
    </div>
  );
}
```

### Marking Item as Complete

```typescript
import { toggleTrackableCompleted } from "~/lib/models/track.server";
import { requireAuth } from "~/lib/services/auth.server";

export async function action({ request }: ActionFunctionArgs) {
  await requireAuth(request);
  
  const formData = await request.formData();
  const trackableId = formData.get("trackableId") as string;
  
  const result = await toggleTrackableCompleted(trackableId, request);
  
  return json(result);
}

export default function TrackableItem({ trackable }) {
  const fetcher = useFetcher();
  
  const handleToggle = () => {
    fetcher.submit(
      { trackableId: trackable.pivot.id },
      { method: "post" }
    );
  };
  
  return (
    <div>
      <h3>{trackable.name}</h3>
      <button onClick={handleToggle}>
        {trackable.completed ? "Mark Incomplete" : "Mark Complete"}
      </button>
    </div>
  );
}
```
