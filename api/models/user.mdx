## Functions

### impersonate

Allows an admin to impersonate another user for support purposes.

<ParamField path="request" type="any" required>
  Request object with admin authentication
</ParamField>

<ParamField path="userId" type="string" required>
  ID of the user to impersonate
</ParamField>

```typescript
async function impersonate(
  request: any,
  userId: string,
): Promise<string | null>;
```

**Returns**: Authentication token for the impersonated user, or null on error

**Example**:
```typescript
import { impersonate } from "~/lib/models/user.server";
import { requireAuth } from "~/lib/services/auth.server";

export async function action({ request }: ActionFunctionArgs) {
  const user = await requireAuth(request);
  
  // Only admins can impersonate
  if (!user.is_admin) {
    throw new Response("Forbidden", { status: 403 });
  }
  
  const formData = await request.formData();
  const userId = formData.get("userId") as string;
  
  const token = await impersonate(request, userId);
  
  if (!token) {
    return json({ error: "Failed to impersonate user" }, { status: 400 });
  }
  
  // Store the impersonation token and redirect
  return redirect("/dashboard", {
    headers: {
      "Set-Cookie": `impersonation_token=${token}; Path=/; HttpOnly`,
    },
  });
}
```

## Usage Examples

### Accessing Current User

```typescript
import { requireAuth } from "~/lib/services/auth.server";
import type { User } from "~/lib/models/user.server";

export async function loader({ request }: LoaderFunctionArgs) {
  const user = await requireAuth(request);
  
  return json({ user });
}

export default function ProfilePage() {
  const { user } = useLoaderData<typeof loader>();
  
  return (
    <div>
      <img src={user.avatar.avatar_url} alt={user.name} />
      <h1>{user.name}</h1>
      <p>{user.email}</p>
      {user.is_pro && <span>PRO Member</span>}
      {user.is_admin && <span>Admin</span>}
    </div>
  );
}
```

### Using User Data from Outlet Context

```typescript
import { useOutletContext } from "react-router";
import type { User } from "~/lib/models/user.server";

export default function SomePage() {
  const { user } = useOutletContext<{ user: User }>();
  
  return (
    <div>
      <p>Welcome, {user.name}!</p>
      {user.is_pro ? (
        <p>You have access to all PRO content</p>
      ) : (
        <p>Upgrade to PRO for full access</p>
      )}
    </div>
  );
}
```

### Displaying User Avatar with Badge

```typescript
import type { UserAvatar } from "~/lib/models/user.server";

function AvatarComponent({ avatar }: { avatar: UserAvatar }) {
  return (
    <div className="relative">
      <img
        src={avatar.avatar_url}
        alt={avatar.name}
        className="rounded-full"
      />
      {avatar.badge && (
        <span className="badge">
          {avatar.badge === "pro" ? "PRO" : "ADMIN"}
        </span>
      )}
    </div>
  );
}
```

### Checking User Permissions

```typescript
import { requireAuth } from "~/lib/services/auth.server";

export async function loader({ request }: LoaderFunctionArgs) {
  const user = await requireAuth(request);
  
  // Check if user has PRO access
  if (!user.is_pro) {
    throw redirect("/subscribe");
  }
  
  // Check admin access
  if (!user.is_admin) {
    throw new Response("Forbidden", { status: 403 });
  }
  
  return json({ user });
}
```

### Displaying Challenge User Submission

```typescript
import type { ChallengeUser } from "~/lib/models/user.server";

function SubmissionCard({ challengeUser }: { challengeUser: ChallengeUser }) {
  return (
    <div>
      <div>
        <img src={challengeUser.avatar.avatar_url} alt={challengeUser.user.name} />
        <h3>{challengeUser.user.name}</h3>
        {challengeUser.avatar.badge && (
          <span>{challengeUser.avatar.badge.toUpperCase()}</span>
        )}
      </div>
      
      {challengeUser.submission_image_url && (
        <img src={challengeUser.submission_image_url} alt="Submission" />
      )}
      
      {challengeUser.submission_url && (
        <a href={challengeUser.submission_url} target="_blank" rel="noopener">
          View Live Project
        </a>
      )}
      
      {challengeUser.completed && challengeUser.certificate && (
        <div>Certificate Available</div>
      )}
      
      <div>
        <p>Comments: {challengeUser.comments.length}</p>
        <p>Submitted: {new Date(challengeUser.submitted_at).toLocaleDateString()}</p>
      </div>
    </div>
  );
}
```
