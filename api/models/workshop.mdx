## Overview

The Workshop model manages educational workshop content on Codante.io, including lessons, instructor information, user progress tracking, and related challenges.

## Type Definitions

### WorkshopUser

Represents a user's enrollment and progress in a workshop.

```typescript
type WorkshopUser = {
  id: number;

  completed_at: string;
  created_at: string;
  updated_at: string;
  certificate: Certificate;
};
```

## Functions

### getWorkshops

Fetches a list of workshops with optional technology filter.

<ParamField path="tech" type="string" required>
  Technology filter (empty string for all workshops)
</ParamField>

```typescript
async function getWorkshops({
  tech,
}: {
  tech: string;
}): Promise<Array<WorkshopCard>>;
```

**Returns**: Array of workshop cards

**Example**:
```typescript
import { getWorkshops } from "~/lib/models/workshop.server";

export async function loader({ request }: LoaderFunctionArgs) {
  const url = new URL(request.url);
  const tech = url.searchParams.get("tech") || "";
  
  const workshops = await getWorkshops({ tech });
  
  return { workshops };
}
```

### getWorkshop

Fetches a single workshop by slug.

<ParamField path="slug" type="string" required>
  Workshop slug identifier
</ParamField>

<ParamField path="request" type="any" required>
  Request object for authentication
</ParamField>

```typescript
async function getWorkshop(
  slug: string,
  request: any,
): Promise<Workshop | null>;
```

**Returns**: Workshop object or null if not found

**Example**:
```typescript
import { getWorkshop } from "~/lib/models/workshop.server";

export async function loader({ params, request }: LoaderFunctionArgs) {
  const workshop = await getWorkshop(params.slug, request);
  
  if (!workshop) {
    throw new Response("Not Found", { status: 404 });
  }
  
  return { workshop };
}
```

### userJoinedWorkshop

Enrolls the current user in a workshop.

<ParamField path="slug" type="string" required>
  Workshop slug
</ParamField>

<ParamField path="request" type="any" required>
  Request object for authentication
</ParamField>

```typescript
async function userJoinedWorkshop(
  slug: string,
  request: any,
): Promise<{ message?: string }>;
```

**Returns**: Success message or error

**Example**:
```typescript
import { userJoinedWorkshop } from "~/lib/models/workshop.server";

export async function action({ params, request }: ActionFunctionArgs) {
  const result = await userJoinedWorkshop(params.slug, request);
  
  if (!result) {
    return json({ error: "Failed to join workshop" }, { status: 400 });
  }
  
  return json(result);
}
```

## Usage Examples

### Listing Workshops with Filter

```typescript
import { getWorkshops } from "~/lib/models/workshop.server";

export async function loader({ request }: LoaderFunctionArgs) {
  const url = new URL(request.url);
  const techFilter = url.searchParams.get("tecnologia") || "";
  
  const workshops = await getWorkshops({ tech: techFilter });
  
  return json({ workshops });
}

export default function WorkshopsPage() {
  const { workshops } = useLoaderData<typeof loader>();
  
  return (
    <div>
      {workshops.map((workshop) => (
        <WorkshopCard key={workshop.id} workshop={workshop} />
      ))}
    </div>
  );
}
```

### Workshop Detail Page

```typescript
import { getWorkshop } from "~/lib/models/workshop.server";
import { requireAuth } from "~/lib/services/auth.server";

export async function loader({ params, request }: LoaderFunctionArgs) {
  await requireAuth(request);
  
  const workshop = await getWorkshop(params.slug, request);
  
  if (!workshop) {
    throw new Response("Workshop not found", { status: 404 });
  }
  
  return json({ workshop });
}

export default function WorkshopDetailPage() {
  const { workshop } = useLoaderData<typeof loader>();
  
  return (
    <div>
      <h1>{workshop.name}</h1>
      <p>{workshop.description}</p>
      <p>Difficulty: {workshop.difficulty}</p>
      <p>Instructor: {workshop.instructor.name}</p>
      <p>Lessons: {workshop.lessons.length}</p>
    </div>
  );
}
```

### Enrolling in a Workshop

```typescript
import { userJoinedWorkshop } from "~/lib/models/workshop.server";
import { requireAuth } from "~/lib/services/auth.server";

export async function action({ params, request }: ActionFunctionArgs) {
  await requireAuth(request);
  
  const result = await userJoinedWorkshop(params.slug, request);
  
  if (!result?.message) {
    return json(
      { error: "Could not join workshop" },
      { status: 400 }
    );
  }
  
  return json({ success: true, message: result.message });
}
```
